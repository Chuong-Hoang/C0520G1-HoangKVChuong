★ Bài tập 1: Chương trình nhập điểm sinh viên

    let physics = prompt("Nhập điểm môn Lý:");
    let chemistry = prompt("Nhập điểm môn Hóa:");
    let math = prompt("Nhập điểm môn Toán:");
    let physMark = parseInt(physics);
    let chemMark = parseInt(chemistry);
    let mathMark = parseInt(math);
    let sum = (physMark + chemMark + mathMark);
    let avg = sum/3;
    document.write("Tổng điểm 3 môn là: " + sum);
    document.write("Điểm trung bình là: " + avg);

★ Bài tập 2: Chương trình chuyển đổi nhiệt độ

    let celsius = prompt("Nhập nhiệt độ thang Celsius:");
    let celsiusDeg = parseInt(celsius);
    let Fahr = (celsius*9/5) + 32;
    document.write("Nhiệt độ thang Celsius là: " + celsiusDeg);
    document.write("Nhiệt độ thang Fahrenheit là: " + Fahr);

");

");
