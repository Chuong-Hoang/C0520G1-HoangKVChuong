Bài 5: Xếp hạng học lực của học sinh

    let test = parseFloat(prompt("Input the test mark: "));
    let midExam = parseFloat(prompt("Input the mid-examination mark: "));
    let exam = parseFloat(prompt("Input the examination mark: "));
    let average = (test + (2*midExam) + (4*exam))/7;
    let msg = '';
    if (average >= 9) {
        msg = "Excellent";
    } else if (average >= 8) {
       msg = "Very good";
    } else if (average >= 7) {
        msg = "Good";
    } else if (average >= 6) {
       msg = "Acceptable";
    } else if(average >= 5) {
        msg = "Not bad";
    } else {
       msg = "Very bad!";
    }
    alert("Average Mark: " + average + "." + "\n" + "Classification: " + msg);

Bài 6: Tính hoa hồng nhận được tùy theo doanh số bán hàng

    let revenue = parseFloat(prompt("Total monthly revenue is: "));
    let commission = 0;
    let rate = 0;
    if (revenue >= 50000) {
        rate = 0.2;
    } else if (revenue >=40000) {
        rate = 0.15;
    } else if (revenue >=30000) {
        rate = 0.1;
    } else if (revenue >= 20000) {
        rate = 0.05;
    } else {
        rate = 0;
    }
    commission = rate * revenue;
    alert("Your commission is: " + commission);

Bài 7: Tính cước điện thoại cho 1 hộ gia đình

    let localNetTime = parseInt(prompt("Local network calling time: "));
    let otherNetTime = parseInt(prompt("Other network calling time: "));
    let overseaTime = parseInt(prompt("Oversea network calling time: "));
    let price = 0;
    let amount = 0;
    //amount for local network calling time
    if (localNetTime >= 0) {
        price = 1200;
        amount += (localNetTime*price);
    }
    // added amount for other network calling time
    if (otherNetTime >= 0) {
        price = 2000;
        amount += (otherNetTime*price);
    }
    // added amount for oversea network calling time
    if (overseaTime >= 0) {
        price = 4000;
        amount += (overseaTime*price);
    }
    alert("Total amount: " + amount);
